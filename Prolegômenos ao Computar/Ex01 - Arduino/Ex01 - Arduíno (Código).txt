int flag = 1; // criamos uma variável flag

void setup() // executa uma vez só
{
  pinMode(13, OUTPUT);
  pinMode(2, INPUT);
}

void loop() // executa em loop eternamente
{
  if(digitalRead(2) == HIGH)
  {
      flag++; // cada loop a flag aumenta 1 unidade
  }
  if(flag%2 == 0) // se a flag for par o led acende
  { 
    digitalWrite(13, HIGH);
    delay(100);
  }  
  else // se a flag for impar o led apaga
  {
    digitalWrite(13, LOW);
    delay(100);
  }   
}
